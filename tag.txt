git tag 'lesson_TAG'   		- создать тег для текущего коммита
git tag          			- перечисляет метки в алфавитном порядке
git tag -l 'v1*' 			- список по маске
git show tag_name 

git tag : tag_name       	- удалить тег на удалёнке
git push --tags origin   	- закачать все теги на удалёнку
git tag -d                  - удаляет на локали все теги
